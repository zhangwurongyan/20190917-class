``` r
title_vec = hpsaga_html %>% 
  html_nodes(".lister-item-header a") %>% 
  html_text()

  hpsaga_html %>%
  html_nodes(".text-small:nth-child(7) span:nth-child(5)") %>%
  html_text()

  hpsaga_html %>%
  html_nodes(".runtime") %>%
  html_text()

  tibble(
```
